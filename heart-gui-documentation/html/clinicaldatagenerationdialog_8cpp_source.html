327 QStringList ClinicalDataGenerationDialog::filterPatientList(QStringList list, QStringList inclusion, QStringList exclusion,
328  RulesValues _type)
329 {
330  // Note: 1st select all those the user want to be included, and then from the included one, I remove those in the exclusion list.
331  // If no inclusion criteria is written, the user is including all teh files, except those with exclusion criteria
332  QStringList _folderList;
333  if (list.isEmpty())
334  return _folderList;
335 
336 
337  QStringList _tmpIncludeList = list;
338  for (int j= 0; j < inclusion.size();j++)
339  {
340  QString _inclVal = inclusion.at(j);
341 
342  if (_inclVal.isEmpty())
343  continue;
344  _tmpIncludeList = filterIncludedValues(_tmpIncludeList,_inclVal,_type);
345  }
346 
347 
348  QStringList _tmpExclList = _tmpIncludeList;
349  for (int i= 0; i < exclusion.size();i++)
350  {
351  QString _exclVal = exclusion.at(i);
352 
353  if (_exclVal.isEmpty())
354  continue;
355  _tmpExclList = filterExcludedValues(_tmpExclList,_exclVal,_type);
356  }
357 
358  _folderList = _tmpExclList;
359  return _folderList;
360 
361 }

403 QList<PatientData> ClinicalDataGenerationDialog::completePatientData(QList<PatientData> _list, QStringList _inclusionPrefixRules,
404  QStringList _iclusionSuffixRules,
405  QStringList _exclusionPrefixRules, QStringList _exclusionSuffixRules)
406 {
407  QDir _cDirectory;

411  return _newList;
412

414  {
415  PatientData _cPatient = _list.at(i);
416  QString _patientPath = QDir(_cPath).filePath(_cPatient.getPatientName());
417  // Saving the Patient Folder Path
418  _cPatient.setFolderPatientPath(_patientPath);

421  QStringList _allFiles = _cDirectory.entryList(QDir::Files);
422  QStringList _listPreffix = _allFiles;
423  QStringList _listSuffix;

425  // Filter files

427  _listPreffix = filterPatientList(_allFiles,_inclusionPrefixRules,_exclusionPrefixRules,PREFFIX_RULES);
428  _listSuffix = _listPreffix;
429  if (!_iclusionSuffixRules.isEmpty())
430  _listSuffix = filterPatientList(_listSuffix,_iclusionSuffixRules,_exclusionSuffixRules,SUFFIX_RULES);
431 
432  QString _fileToShow;
433  if ((!_listSuffix.isEmpty()) && (_listSuffix.size()>=1))
434  _fileToShow = _listSuffix.at(0);
435  // Saving the Image File Path
436  _cPatient.setFileImagePath(_fileToShow);
437 
438  _newList.append(_cPatient);
439 
440  }
441  return _newList;

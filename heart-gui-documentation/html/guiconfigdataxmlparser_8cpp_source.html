218 void GuiConfigDataXMLParser::readMatlabConfiguration( QXmlStreamReader * xmlReader,GUIConfigData * _userData)
219 {
220  bool _endItem = false;

244  _data.setNoSplash(true);
245  else
246  _data.setNoSplash(false);
247 
248  }
249 
250  // Scripts
251  if ((token == QXmlStreamReader::StartElement) &&
252  (QString::compare(xmlReader->name().toString(), USER_GUI_MATLAB_SCRIPT_CONFIG_TAG) == StrEqual) )
253  {
254  QXmlStreamAttributes _elementAtt = xmlReader->attributes();
255  QString _path = "";
256  QString _nameScript = "";
257  if (_elementAtt.hasAttribute(USER_GUI_MATLAB_SCRIPT_CONFIG_PATH_ATT))

259

265  }

267  (QString::compare(xmlReader->name().toString(), USER_GUI_MATLAB_CONFIG_TAG) == StrEqual) )
268  _endItem = true;
269  }
270 
271  _data.setScriptsPath(_scriptList);
272  _userData->setMatlabData(_data);
